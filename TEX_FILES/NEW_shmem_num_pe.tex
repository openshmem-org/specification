\bAPI{SHMEM\_NUM\_\ac{PE}}{Returns processing element (\ac{PE}) information.}

%Synopsis C
\synC
	  int shmem_n_pes(void);
	  int _num_pes (void);
%*\synCE

%Synopsis F
\synF
	  INTEGER SHMEM_N_\ac{PE}S, N_\ac{PE}S
	  N\ac{PE}S = SHMEM_N_\ac{PE}S()
	  N_\ac{PE}S = NUM_\ac{PE}S()
%*\synFE

%DESCRIPTION

%Arguments
\desB{
	\argRow{NONE}{}{}
}
%API Description
{
	The function returns the number of \ac{PE}s running in an application.
}
%API Description Table.
{
		\desTB{}
		{
				\cRow{}{}
		}
		%Return Value
		\desR{Integer -  Number  of  \ac{PE}s  running the SPMD application.}
		%NOTES      
		\notesB{
				For OpenSHMEM Specification 1.1 the use of \_my\_pe() has been deprecated. Although OpenSHMEM libraries are required to support the call, application developers are encouraged to use shmem\_my\_pe() instead.
		}
}
