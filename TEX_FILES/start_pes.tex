\bAPI{START\_PES}{Called at the beginning of an \openshmem program to initialize the execution environment.}
\synC   %Synopisis for C API 

void start_pes(int npes); %*\synCE    %DO NOT DELETE. THIS LINE IS NOT A COMMENT

\synF   %Synopsis for FORTRAN API

CALL START_PES(npes) %*\synFE   %DO NOT DELETE. THIS LINE IS NOT A COMMENT  

% Arguments table. If no arguments you can use \argRow{None}{}{} 
\desB{  
       \argRow{npes}{Unused}{ Should be set to \CONST{0}.}
}
 %API description
 {   
     The \FUNC{start\_pes} routine initializes the \openshmem execution environment.  An \openshmem application must call \FUNC{start\_pes} before calling any other \openshmem routine.
 }
 %API Description Table. 
{
 %Return Values     
\desR{ None. }
\notesB{ If any other \openshmem call occurs before \FUNC{start\_pes}, the behavior is undefined.
Although it is recommended to set \VAR{npes} to \CONST{0}, this is not mandated.  The value is ignored.
Calling \FUNC{start\_pes} more than once has no subsequent effect.
}
}%end of DesB
