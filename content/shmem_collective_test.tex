\apisummary{
    The routine outputs the status of the operation identified by the request.
}

\begin{apidefinition}

\begin{Csynopsis}
int @\FuncDecl{shmem\_req\_test}@(shmem_req_h request);
\end{Csynopsis}

\begin{apiarguments}

  \apiargument{IN}{request}{Request handle}

\end{apiarguments}

\apidescription{
    A call to \FUNC{shmem\_req\_test} returns immediately. If the
    operation identified by the request is completed, it returns
    zero, and the request object is deallocated. If the operation is not
    completed, it returns a non-negative integer.

    In a multithreaded environment, the operation identified by the request and the
    \FUNC{shmem\_req\_test} can be
    called by different threads. It is the responsibility of the \openshmem user
    to ensure that the \FUNC{shmem\_req\_test} operation is called after the
    operation has been initiated.
    }

\apireturnvalues{
    On success returns zero, otherwise returns a negative integer.
    }

\end{apidefinition}
