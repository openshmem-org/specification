\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_allocate(shmem_request_t *req, int options);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN/OUT}{req}{Handle to the request object.}
    \apiargument{IN}{options}{Options for the request creation.}
\end{apiarguments}

\apidescription{
    This routine can be used to explicitly pre-allocate a request
    object. On failure the handle will be set to SHMEM\_REQUEST\_INVALID.
    The flags parameter can be used to pass hints to the
    runtime.\\

    Options:\\
    OPTNAME\_EXCLUSIVE: Allocate the request with a dedicated set of resources.
    Fail if exclusive resources can not be provided.\\
    OPTNAME\_EXCLUSIVE\_TRY: Allocate the request with a dedicated set of resources.
    Return a request with shared resources, if dedicated resources can
    not be allocated.\\
    OPTNAME\_SHAREABLE: Allocate the request with shareable resources.\\

 }

\apireturnvalues{
    None.
}

\apinotes{ None }

\end{apidefinition}
