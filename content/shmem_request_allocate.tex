\begin{apidefinition}

\begin{Csynopsis}
void shmem_request_allocate(shmem_request_t *req, int options);
\end{Csynopsis}

\begin{apiarguments}
    \apiargument{IN/OUT}{req}{Handle to the request object.}
    \apiargument{IN}{options}{Options for the request creation.}
\end{apiarguments}

\apidescription{
    This routine can be used to explicitly pre-allocate a request
    object. On failure the handle will be set to SHMEM\_REQUEST\_INVALID.
    The flags parameter can be used to pass hints to the
    runtime.\\

    Options:\\
    SHMEM_REQUEST\_PRIVATE: Allocate a private request.
    The request can not be used from multiple threads if allocated with
    this option. \\
    SHMEM\_REQUEST\_SHARED: Allocate a shareble request. Sharable
    requests can safely be used from multiple threads. \\
 }

\apireturnvalues{
    None.
}

\apinotes{ None }

\end{apidefinition}
