\bAPI{SHMEM\_N\_PES}{Returns the number of \ac{PE}s running in a program.}
%Synopsis C
\synC
int shmem_n_pes(void); %*\synCE

%Synopsis F
\synF
INTEGER SHMEM_N_PES, N_PES

%DESCRIPTION

\desB{
	\argRow{None}{}{}
}
